MarkUp:

	<li>
		<a href="#">Item 1</a>
	</li>
	<li>
		<a href="#">Item 2</a>
		<ul>
			<li>
				<a href="#">Item 2-1</a>
			</li>
			<li>
				<a href="#">Item 2-2</a>

</ul>

Usage:

$(document).ready(function(){
	$('#amenu-list').amenu({
		'speed': 200,
		'animation': 'wind'       //  show, fade, slide, wind, none
	});
});
